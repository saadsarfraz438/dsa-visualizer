What is a Linked List?

A linked list is a linear data structure where elements (nodes) are stored non-contiguously in memory. Each node contains data and a pointer to the next node, forming a chain.

Key Properties

  • Dynamic size — grows and shrinks at runtime
  • Nodes need not be stored contiguously
  • Efficient insertion/deletion at any position
  • No random access — must traverse sequentially

Applications

  • Implementation of stacks and queues
  • Undo/redo operations in editors
  • Adjacency lists in graphs
  • Polynomial arithmetic

Types

  • Singly Linked List — one direction pointer
  • Doubly Linked List — forward and backward
  • Circular Linked List — last node → head
C++
// Singly Linked List in C++
#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* next;
    Node(int val) : data(val), next(nullptr) {}
};

class LinkedList {
    Node* head;
public:
    LinkedList() : head(nullptr) {}

    // Insert at beginning  O(1)
    void insertFront(int val) {
        Node* node = new Node(val);
        node->next = head;
        head = node;
    }

    // Insert at end  O(n)
    void insertEnd(int val) {
        Node* node = new Node(val);
        if (!head) { head = node; return; }
        Node* cur = head;
        while (cur->next) cur = cur->next;
        cur->next = node;
    }

    // Delete by value  O(n)
    void deleteVal(int val) {
        if (!head) return;
        if (head->data == val) {
            Node* tmp = head;
            head = head->next;
            delete tmp; return;
        }
        Node* cur = head;
        while (cur->next && cur->next->data != val)
            cur = cur->next;
        if (cur->next) {
            Node* tmp = cur->next;
            cur->next = tmp->next;
            delete tmp;
        }
    }

    // Search  O(n)
    bool search(int val) {
        Node* cur = head;
        while (cur) {
            if (cur->data == val) return true;
            cur = cur->next;
        }
        return false;
    }

    void display() {
        Node* cur = head;
        while (cur) {
            cout << cur->data << " → ";
            cur = cur->next;
        }
        cout << "NULL\n";
    }
};

What is a Stack?

A stack is an abstract data type that follows LIFO (Last In, First Out). Think of a pile of plates — you add and remove from the top only. It restricts access to only the topmost element.

Core Operations

  • push(x) — Add element to top
  • pop() — Remove and return top element
  • peek() — View top without removing
  • isEmpty() — Check if stack has elements

Applications

  • Function call stack (recursion)
  • Expression evaluation (postfix)
  • Undo/redo in software
  • Backtracking algorithms
  • Balanced parentheses checker

Implementations

  • Array-based stack (fixed size)
  • Linked list stack (dynamic)
  • C++ STL: #include <stack>
C++
// Stack implementation in C++
#include <iostream>
#define MAX 1000
using namespace std;

class Stack {
    int arr[MAX];
    int top;
public:
    Stack() : top(-1) {}

    bool push(int x) {
        if (top >= MAX - 1) {
            cout << "Stack Overflow\n";
            return false;
        }
        arr[++top] = x;
        return true;
    }

    int pop() {
        if (top < 0) {
            cout << "Stack Underflow\n";
            return -1;
        }
        return arr[top--];
    }

    int peek() {
        if (top < 0) return -1;
        return arr[top];
    }

    bool isEmpty() { return top < 0; }
    int  size()    { return top + 1; }

    void display() {
        cout << "Stack (top→bottom): ";
        for (int i = top; i >= 0; i--)
            cout << arr[i] << " ";
        cout << "\n";
    }
};

What is a Queue?

A queue is a linear data structure following FIFO (First In, First Out). Think of a line at a ticket counter — the person who arrives first is served first. Elements enter at the rear and leave from the front.

Core Operations

  • enqueue(x) — Add element at rear
  • dequeue() — Remove element from front
  • front() — Peek at front element
  • rear() — Peek at rear element

Applications

  • CPU / disk scheduling
  • BFS graph traversal
  • Print spooler
  • Asynchronous data transfer (buffers)
  • Call center systems

Types

  • Simple Queue — basic FIFO
  • Circular Queue — avoids wasted space
  • Double Ended Queue — insert/delete both ends
  • Priority Queue — ordered by priority
C++
// Queue implementation in C++
#include <iostream>
#define MAX 1000
using namespace std;

class Queue {
    int arr[MAX];
    int front, rear, sz;
public:
    Queue() : front(0), rear(-1), sz(0) {}

    bool enqueue(int x) {
        if (sz == MAX) {
            cout << "Queue Full\n"; return false;
        }
        arr[++rear] = x; sz++;
        return true;
    }

    int dequeue() {
        if (sz == 0) {
            cout << "Queue Empty\n"; return -1;
        }
        sz--;
        return arr[front++];
    }

    int getFront()  { return sz ? arr[front] : -1; }
    int getRear()   { return sz ? arr[rear]  : -1; }
    bool isEmpty() { return sz == 0; }
    int  size()    { return sz; }

    void display() {
        cout << "Front → ";
        for (int i = front; i <= rear; i++)
            cout << arr[i] << " ";
        cout << "← Rear\n";
    }
};

Binary Search Tree Properties

  • Every node has at most 2 children
  • Left subtree values are less than parent
  • Right subtree values are greater than parent
  • No duplicate keys
  • Each subtree is itself a valid BST

Traversal Types

  • Inorder — Left, Root, Right → sorted order
  • Preorder — Root, Left, Right → copy tree
  • Postorder — Left, Right, Root → delete tree
  • Level Order — BFS breadth-first

Applications

  • Dynamic set operations
  • Database indexing (B-trees)
  • Auto-complete / spell checker
  • File system hierarchies

Balance Note

An unbalanced BST degrades to O(n) in the worst case (like a linked list). Self-balancing variants like AVL and Red-Black trees maintain O(log n) by rotation.

C++
// Binary Search Tree in C++
#include <iostream>
#include <queue>
using namespace std;

struct Node {
    int val;
    Node *left, *right;
    Node(int v) : val(v), left(nullptr), right(nullptr) {}
};

class BST {
    Node* root;

    Node* insert(Node* node, int v) {
        if (!node) return new Node(v);
        if (v < node->val) node->left  = insert(node->left, v);
        else if (v > node->val) node->right = insert(node->right, v);
        return node;
    }

    Node* minNode(Node* node) {
        while (node->left) node = node->left;
        return node;
    }

    Node* deleteNode(Node* node, int v) {
        if (!node) return node;
        if (v < node->val) node->left  = deleteNode(node->left, v);
        else if (v > node->val) node->right = deleteNode(node->right, v);
        else {
            if (!node->left) { Node* t = node->right; delete node; return t; }
            if (!node->right){ Node* t = node->left;  delete node; return t; }
            Node* s = minNode(node->right);
            node->val = s->val;
            node->right = deleteNode(node->right, s->val);
        }
        return node;
    }

    void inorder(Node* n) {
        if (!n) return;
        inorder(n->left);
        cout << n->val << " ";
        inorder(n->right);
    }

public:
    BST() : root(nullptr) {}
    void insert(int v)      { root = insert(root, v); }
    void remove(int v)      { root = deleteNode(root, v); }
    void printInorder()    { inorder(root); cout << "\n"; }
};

What is a Graph?

A graph G = (V, E) consists of a set of vertices V and edges E. Edges connect pairs of vertices and can be directed or undirected, weighted or unweighted.

BFS vs DFS

  • BFS — uses queue, explores level-by-level, shortest path
  • DFS — uses stack/recursion, explores depth-first
  • BFS: O(V+E) time, O(V) space
  • DFS: O(V+E) time, O(V) call stack

Representations

  • Adjacency Matrix — O(V²) space, O(1) edge lookup
  • Adjacency List — O(V+E) space, efficient for sparse
  • Edge List — simplest, for Kruskal's algorithm

Applications

  • Social networks (friend suggestions)
  • GPS navigation (shortest path)
  • Compiler dependency resolution
  • Network routing protocols
C++
// Graph using Adjacency List in C++
#include <iostream>
#include <list>
#include <queue>
#include <vector>
using namespace std;

class Graph {
    int V;
    vector<list<int>> adj;
public:
    Graph(int v) : V(v), adj(v) {}

    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u); // undirected
    }

    // BFS — O(V + E)
    void BFS(int src) {
        vector<bool> visited(V, false);
        queue<int> q;
        visited[src] = true;
        q.push(src);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            cout << v << " ";
            for (int u : adj[v]) {
                if (!visited[u]) {
                    visited[u] = true;
                    q.push(u);
                }
            }
        }
    }

    // DFS helper — O(V + E)
    void DFSHelper(int v, vector<bool>& vis) {
        vis[v] = true;
        cout << v << " ";
        for (int u : adj[v])
            if (!vis[u]) DFSHelper(u, vis);
    }

    void DFS(int src) {
        vector<bool> visited(V, false);
        DFSHelper(src, visited);
    }
};
